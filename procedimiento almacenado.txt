CREATE OR REPLACE FUNCTION INGRESOS_OBTENIDO(IN SERVIC integer,
									OUT INGRESOS_OBTENIDO  varchar)
as 
$$
begin 
--consulta
	SELECT 
	SUM(SERVICIO.SER_COSTO) INTO INGRESOS_OBTENIDO
	FROM RESERVA_DETALLE
	INNER JOIN SERVICIO ON RESERVA_DETALLE.VELATORIO_ID=SERVICIO.SER_ID
	WHERE RESERVA_DETALLE.VELATORIO_ID=SERVIC;
end
$$
language 'plpgsql';
SELECT * FROM INGRESOS_OBTENIDO(4)
